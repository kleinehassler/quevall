   FROM sujetos s
   LEFT JOIN detagru c ON s.sciudad = c.iddato
   LEFT JOIN detagru u ON s.sciudad2 = u.iddato
   LEFT JOIN detagru p ON s.sprov = p.iddato
   LEFT JOIN detagru t ON s.tipcli = t.iddato
   LEFT JOIN detagru r ON s.parroquia = r.iddato
   LEFT JOIN detagru e ON s.sector = e.iddato
   LEFT JOIN detagru x on s.tipdocextraj=x.iddato
   LEFT JOIN rutas f ON s.sector = f.ruta
   LEFT JOIN sujetos v ON f.vendedor = v.idsujeto
   LEFT JOIN sujetos b ON s.codprin = b.idsujeto;;

CREATE OR REPLACE VIEW vadqing AS 
 SELECT a.iditem, a.fecha, a.punitario, a.costoprom
   FROM (         SELECT s.iditem, p.fecini AS fecha, s.cospro AS punitario, s.precio AS costoprom
                   FROM saldosi s
              LEFT JOIN periodos p ON s.fecsaldo = p.fecini
              where not fecini isnull
        UNION all
                 SELECT t.iditem, d.fecha, max(t.punitario - t.descuento / t.qty) AS punitario, max(t.costoprom) AS costoprom
                   FROM detadoci t
              LEFT JOIN vdocusmall d ON t.code = d.code
             WHERE d.isaccount AND NOT d.isanulado AND (d.iddocu IN ( SELECT DISTINCT d.iddocu
                      FROM gdoc g, dgdoc t, docuse d
                     WHERE g.tag::text = 'ADQING'::text AND g.idgdoc = t.idgdoc AND t.dtag::text = d.dtag::text))
             GROUP BY t.iditem, d.fecha) a;

   FROM (         SELECT s.iditem, p.fecini AS fecha, s.cospro AS punitario, s.precio AS costoprom, qtyini as qty, bodega
                   FROM saldosi s LEFT JOIN periodos p ON s.fecsaldo = p.fecini
				  left join saldosib b on s.idsaldosi=b.idsaldosi
              where not fecini isnull
        UNION all
                 SELECT t.iditem, d.fecha, ((t.punitario*t.qty - t.descuento) / t.qty) AS punitario, t.costoprom AS costoprom,

                   FROM detadoci t
              LEFT JOIN vdocusmall d ON t.code = d.code
             WHERE d.isaccount AND NOT d.isanulado AND (d.iddocu IN ( SELECT DISTINCT d.iddocu
                      FROM gdoc g, dgdoc t, docuse d
                     WHERE g.tag::text = 'KARKAR'::text AND g.idgdoc = t.idgdoc AND t.dtag::text = d.dtag::text)) ) a ;
